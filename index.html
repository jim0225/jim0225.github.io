from flask import Flask, render_template_string import zlib app = Flask(__name__) HTML_TEMPLATE = ''' 데이터 압축 테스트

데이터 압축 테스트

{% for result in results %}

{{ result['name'] }}

원본 크기: {{ result['original_size'] }} bytes

압축된 크기: {{ result['compressed_size'] }} bytes

압축률: {{ result['rate'] }}%

{% endfor %}
''' @app.route('/') def index(): size = 1000 # 데이터 크기 repeats = 1 # 반복 횟수 results = [] # 규칙적인 데이터 pattern_data = b'01' * (size // 2 * repeats) compressed = zlib.compress(pattern_data) results.append({ 'name': '규칙적인 데이터', 'original_size': len(pattern_data), 'compressed_size': len(compressed), 'rate': f"{(len(compressed) / len(pattern_data) * 100):.2f}" }) # 자연어 데이터 natural_text = "The quick brown fox jumps over the lazy dog. " * repeats natural_data = natural_text.encode('utf-8')[:size] compressed = zlib.compress(natural_data) results.append({ 'name': '자연어 데이터', 'original_size': len(natural_data), 'compressed_size': len(compressed), 'rate': f"{(len(compressed) / len(natural_data) * 100):.2f}" }) # 무작위 데이터 import random random_data = bytes(random.randint(0, 255) for _ in range(size * repeats)) compressed = zlib.compress(random_data) results.append({ 'name': '무작위 데이터', 'original_size': len(random_data), 'compressed_size': min(len(compressed), size * repeats), 'rate': f"{(min(len(compressed), size * repeats) / (size * repeats) * 100):.2f}" }) return render_template_string(HTML_TEMPLATE, results=results) if __name__ == '__main__': app.run(debug=True)
